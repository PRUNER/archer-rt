# CMakeLists.txt file for unit testing Sword runtime library.
include(CheckFunctionExists)
include(CheckLibraryExists)

# Some tests use math functions
check_library_exists(m sqrt "" LIBSWORD_HAVE_LIBM)
# When using libgcc, -latomic may be needed for atomics
# (but when using compiler-rt, the atomics will be built-in)
# Note: we can not check for __atomic_load because clang treats it
# as special built-in and that breaks CMake checks
check_function_exists(__atomic_load_1 LIBSWORD_HAVE_BUILTIN_ATOMIC)
if(NOT LIBSWORD_HAVE_BUILTIN_ATOMIC)
  check_library_exists(atomic __atomic_load_1 "" LIBSWORD_HAVE_LIBATOMIC)
else()
  # not needed
  set(LIBSWORD_HAVE_LIBATOMIC 0)
endif()

set(LIBSWORD_TEST_PATH ${CMAKE_CURRENT_SOURCE_DIR})
set(LIBSWORD_HAVE_SWORD_LIBRARY ${LIBSWORD_LIB_PATH})
set(LIBSWORD_HAVE_SWORD_RUNTIME ${LIBSWORD_RUNTIME_PATH})

macro(pythonize_bool var)
  if (${var})
    set(${var} True)
  else()
    set(${var} False)
  endif()
endmacro()

pythonize_bool(LIBSWORD_HAVE_SWORD_LIBRARY)
pythonize_bool(LIBSWORD_HAVE_SWORD_RUNTIME)
pythonize_bool(LIBSWORD_OMPT_SUPPORT)
pythonize_bool(LIBSWORD_OMPT_OPTIONAL)
pythonize_bool(LIBSWORD_HAVE_LIBM)

add_sword_testsuite(check-libsword "Running libsword tests" ${CMAKE_CURRENT_BINARY_DIR} DEPENDS sword LLVMSword)

# Configure the lit.site.cfg.in file
set(AUTO_GEN_COMMENT "## Autogenerated by libsword configuration.\n# Do not edit!")
configure_file(lit.site.cfg.in lit.site.cfg @ONLY)
